218 bboxMin = newBBoxMin;

221 bboxDelta = (newBBoxMax - newBBoxMin) / ((iCrdCmpT(1) << maxDepth));
222 if (newBBoxMax <= newBBoxMin) {
223 std::cerr << "ERROR(setBBox): newBBoxMin must be less than newBBoxMax!" << std::endl;
224 exit(1);

227 bboxDelta = newBBoxMax;
228 if ((newBBoxMin.size() != domDim) || (newBBoxMax.size() != domDim)) {
229 std::cerr << "ERROR(setBBox): New bounding box vectors should be of length domDim!" << std::endl;
230 exit(1);

233 bboxDelta[i] = (newBBoxMax[i] - newBBoxMin[i]) / ((iCrdCmpT(1) << maxDepth));

235 std::cerr << "ERROR(setBBox): Corrisponding elements of newBBoxMin must be less than newBBoxMax!" << std::endl;

525 listCrdTup rv;

711 void sampleFunction(iiCrdTupT cell, int depth, std::function<rangeT(domainT)> func) {
712 if ((cellLevel(cell)+depth) < maxDepth) {
713 iiCrdTupT tmp = cellGetCornerMin(cell);
714 iCrdCmpT del = cellHalfWidth(cell) >> depth;
715 if constexpr (domDim == 1) {
716 iCrdCmpT del = cellHalfWidth(cell) >> depth;
717 for(int i=0; i<(1 << (1+depth))+1; i++) {
718 samplePoint(tmp, func);
719 tmp += del;
720 }
721 } else if constexpr (domDim == 2) {
722 iCrdCmpT del = cellFullWidth(cell) >> depth;
723 // Corners
724 for(int i=0; i<(1 << depth)+1; i++) {
725 iiCrdTupT tmp2 = incCrd(tmp, 0, i*del);
726 for(int j=0; j<(1 << depth)+1; j++) {
727 iiCrdTupT tmp3 = incCrd(tmp2, 1, j*del);
728 samplePoint(tmp3, func);
729 }
730 }
731 // Centers
732 tmp = incAllCrd(tmp, del/2);
733 for(int i=0; i<(1 << depth); i++) {
734 iiCrdTupT tmp2 = incCrd(tmp, 0, i*del);
735 for(int j=0; j<(1 << depth); j++) {
736 iiCrdTupT tmp3 = incCrd(tmp2, 1, j*del);
