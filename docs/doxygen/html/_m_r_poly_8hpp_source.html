134 bool seg_isect1(pnt_idx_t ilin1pnt1, pnt_idx_t ilin1pnt2, pnt_idx_t ilin2pnt1, pnt_idx_t ilin2pnt2) {
135 pnt_t lin1pnt1 = pnt_idx_to_pnt[ilin1pnt1];
136 pnt_t lin1pnt2 = pnt_idx_to_pnt[ilin1pnt2];
137 pnt_t lin2pnt1 = pnt_idx_to_pnt[ilin2pnt1];
138 pnt_t lin2pnt2 = pnt_idx_to_pnt[ilin2pnt2];
139 double denom =
140 lin1pnt1[0] * lin2pnt1[1] -
141 lin1pnt1[0] * lin2pnt2[1] -
142 lin1pnt1[1] * lin2pnt1[0] +
143 lin1pnt1[1] * lin2pnt2[0] -
144 lin1pnt2[0] * lin2pnt1[1] +
145 lin1pnt2[0] * lin2pnt2[1] +
146 lin1pnt2[1] * lin2pnt1[0] -
147 lin1pnt2[1] * lin2pnt2[0];
148 if (std::abs(denom) < eps) // Lines are parallel
149 return false;
150 double numera =
151 lin1pnt1[0]*lin2pnt1[1] -
152 lin1pnt1[0]*lin2pnt2[1] -
153 lin1pnt1[1]*lin2pnt1[0] +
154 lin1pnt1[1]*lin2pnt2[0] +
155 lin2pnt1[0]*lin2pnt2[1] -
156 lin2pnt1[1]*lin2pnt2[0];
157 double numerb =
158 -(lin1pnt1[0]*lin1pnt2[1] -
159 lin1pnt1[0]*lin2pnt1[1] -
160 lin1pnt1[1]*lin1pnt2[0] +
161 lin1pnt1[1]*lin2pnt1[0] +
162 lin1pnt2[0]*lin2pnt1[1] -
163 lin1pnt2[1]*lin2pnt1[0]);
164 double ua = numera/denom;
165 double ub = numerb/denom;
166 if (ua < 0)
167 return false;
168 if (ub < 0)
169 return false;
170 if (ua > 1)
171 return false;
172 if (ub > 1)
173 return false;
174 double eq3 = numera/denom * (lin1pnt2[2] - lin1pnt1[2]) - numerb/denom * (lin2pnt2[2] - lin2pnt1[2]) + lin2pnt1[2] + lin1pnt1[2];
175 if (std::abs(eq3) < eps) // Equation in third coordinate is satisfied
176 return true;
177 else
178 return false;
179 }

182 bool pts_colinear(pnt_idx_t pi1, pnt_idx_t pi2, pnt_idx_t pi3) {
183 // P1, P2, P3, P4 = points 1, 2, 3, & 4
184 pnt_t P1 = pnt_idx_to_pnt[pi1];
185 pnt_t P2 = pnt_idx_to_pnt[pi2];
186 pnt_t P3 = pnt_idx_to_pnt[pi3];
187 // P2=P2-P1, P3=P3-P1
188 for(int i=0; i<3; ++i) {
189 P2[i] = P2[i] - P1[i];
190 P3[i] = P3[i] - P1[i];
191 }
192 // P1=P2xP3
193 P1[0] = P2[1]*P3[2]-P2[2]*P3[1];
194 P1[1] = P2[2]*P3[0]-P2[0]*P3[2];
195 P1[2] = P2[0]*P3[1]-P2[1]*P3[0];
196 // If P1=0, then degenerate
197 return ((P1[0]*P1[0]+P1[1]*P1[1]+P1[2]*P1[2])<eps);
198 }
199 //--------------------------------------------------------------------------------------------------------------------------------------------------------
201 bool pts_coplanar(pnt_idx_t pi1, pnt_idx_t pi2, pnt_idx_t pi3, pnt_idx_t pi4) {
202 // P1, P2, P3, P4 = points 1, 2, 3, & 4
203 pnt_t P1 = pnt_idx_to_pnt[pi1];
204 pnt_t P2 = pnt_idx_to_pnt[pi2];
205 pnt_t P3 = pnt_idx_to_pnt[pi3];
206 pnt_t P4 = pnt_idx_to_pnt[pi4];
207 // P2=P2-P1, P3=P3-P1, P4=P4-P1
208 for(int i=0; i<3; ++i) {
209 P2[i] = P2[i] - P1[i];
210 P3[i] = P3[i] - P1[i];
211 P4[i] = P4[i] - P1[i];
212 }
213 // P1=P2xP4
214 P1[0] = P2[1]*P4[2]-P2[2]*P4[1];
215 P1[1] = P2[2]*P4[0]-P2[0]*P4[2];
216 P1[2] = P2[0]*P4[1]-P2[1]*P4[0];
217 // DOTP=P1.P3
218 double DOTP = 0.0;
219 for(int i=0; i<3; ++i)
220 DOTP += std::abs(P1[i] * P3[i]);
221 // If DOTP==0, then degenerate
222 return (DOTP<eps);
223 }

449 cell_stat_t check_cell(cell_type_t cell_type, cell_t a_cell) {
450 // Check number of points
451 int a_cell_len = a_cell.size();
452 int req_num_pts = cell_type_to_cell_size(cell_type);
453 if (a_cell_len < req_num_pts)
454 return cell_stat_t::TOO_FEW_PNT;
455 if (a_cell_len > req_num_pts)
456 return cell_stat_t::TOO_MANY_PNT;
457 // Check for duplicate points
458 std::set<pnt_idx_t> a_cell_pnt_sorted;
459 for(pnt_idx_t pnt_idx: a_cell) {
460 if (a_cell_pnt_sorted.contains(pnt_idx))
461 return cell_stat_t::DUP_PNT;
462 a_cell_pnt_sorted.insert(pnt_idx);
463 }
464 // Check for negative point index
465 if (std::any_of(a_cell.cbegin(), a_cell.cend(), [this](pnt_idx_t i) { return (i < 0); }))
466 return cell_stat_t::NEG_PNT_IDX;
467 // Check for too big point index
468 if (std::any_of(a_cell.cbegin(), a_cell.cend(), [this](pnt_idx_t i) { return (i >= num_points()); }))
469 return cell_stat_t::BIG_PNT_IDX;
470 // Check various degenerate cases
471 if (cell_type == cell_type_t::TRIANGLE) {
472 if (pts_colinear(a_cell[0], a_cell[1], a_cell[2]))
473 return cell_stat_t::TRI_COLIN;
474 } else if (cell_type == cell_type_t::TETRAHEDRON) {
475 if (pts_coplanar(a_cell[0], a_cell[1], a_cell[2], a_cell[3]))
476 return cell_stat_t::TET_COPLN;
477 } else if (cell_type == cell_type_t::PYRAMID) {
478 // Check bent base
479 if ( !(pts_coplanar(a_cell[0], a_cell[1], a_cell[2], a_cell[3])))
480 return cell_stat_t::PYR_BASE_BENT;
481 // Check for colinear sides
482 if (pts_colinear(a_cell[0], a_cell[1], a_cell[2]))
483 return cell_stat_t::PYR_BASE_COLIN;
484 if (pts_colinear(a_cell[1], a_cell[2], a_cell[3]))
485 return cell_stat_t::PYR_BASE_COLIN;
486 if (pts_colinear(a_cell[2], a_cell[3], a_cell[0]))
487 return cell_stat_t::PYR_BASE_COLIN;
488 if (pts_colinear(a_cell[3], a_cell[0], a_cell[1]))
489 return cell_stat_t::PYR_BASE_COLIN;
490 // Check for bow-tie cell_stat_t::PYR_BASE_TIE
491 if (seg_isect1(a_cell[0], a_cell[1], a_cell[2], a_cell[3]))
492 return cell_stat_t::PYR_BASE_TIE;
493 if (seg_isect1(a_cell[1], a_cell[2], a_cell[0], a_cell[3]))
494 return cell_stat_t::PYR_BASE_TIE;
495 // Check cell_stat_t::PYR_COPLN (tip in base plane)
496 if (pts_coplanar(a_cell[0], a_cell[1], a_cell[2], a_cell[4]))
497 return cell_stat_t::PYR_COPLN;
498 }
499 // Everything seems OK
500 return cell_stat_t::GOOD;
501 }
