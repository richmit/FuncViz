159 typename cc_t::pnt_crd_t get_scalar(tree_scl_val_desc_t index, typename rt_t::drpt_t& domainValue, typename rt_t::rrpt_t& rangeValue) {
160 if (std::get<1>(index) == tree_val_src::DOMAIN) {
161 return (rtree.dom_at(domainValue, std::get<2>(index)));
162 } else if (std::get<1>(index) == tree_val_src::RANGE) {
163 return (rtree.rng_at(rangeValue, std::get<2>(index)));

186 tree_vec_val_desc_t const& point_src,
187 tree_scl_val_desc_lst_t const& scalar_data_src_lst,
188 tree_vec_val_desc_lst_t const& vector_data_src_lst) {
189 typename rt_t::drpt_t dPts = rtree.diti_to_drpt(diti);
190 typename rt_t::rrpt_t rPts = rtree.get_sample(diti);
191 typename cc_t::pnt_t pnt = get_vector(point_src, dPts, rPts);
192 typename cc_t::pnt_idx_t pnti = ccplx.add_point(pnt);
193 for (auto s : scalar_data_src_lst)
194 ccplx.add_data_if_new(std::get<0>(s), get_scalar(s, dPts, rPts));
195 for (auto v : vector_data_src_lst)
196 ccplx.add_data_if_new(std::get<0>(v), get_vector(v, dPts, rPts));
197 return (pnti);
198 }
