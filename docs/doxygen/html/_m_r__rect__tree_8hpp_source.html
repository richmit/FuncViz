717 void sample_function(diti_t cell, int depth, std::function<rrpt_t(drpt_t)> func) {
718 if ((cell_level(cell)+depth) < max_depth) {
719 diti_t tmp = cell_get_corner_min(cell);
720 if constexpr (dom_dim == 1) {
721 dic_t del = cell_half_width(cell) >> depth;
722 for(int i=0; i<(1 << (1+depth))+1; i++) {
723 sample_point(tmp, func);
724 tmp += del;
725 }
726 } else if constexpr (dom_dim == 2) {
727 dic_t del = cell_full_width(cell) >> depth;
728 // Corners
729 for(int i=0; i<(1 << depth)+1; i++) {
730 diti_t tmp2 = inc_crd(tmp, 0, i*del);
731 for(int j=0; j<(1 << depth)+1; j++) {
732 diti_t tmp3 = inc_crd(tmp2, 1, j*del);
733 sample_point(tmp3, func);
734 }
735 }
736 // Centers
737 tmp = inc_all_crd(tmp, del/2);
738 for(int i=0; i<(1 << depth); i++) {
739 diti_t tmp2 = inc_crd(tmp, 0, i*del);
740 for(int j=0; j<(1 << depth); j++) {
741 diti_t tmp3 = inc_crd(tmp2, 1, j*del);
742 sample_point(tmp3, func);
743 }
744 }
745 } else if constexpr (dom_dim == 3) {
746 dic_t del = cell_full_width(cell) >> depth;
747 // Corners
748 for(int i=0; i<(1 << depth)+1; i++) {
749 diti_t tmp2 = inc_crd(tmp, 0, i*del);
750 for(int j=0; j<(1 << depth)+1; j++) {
751 diti_t tmp3 = inc_crd(tmp2, 1, j*del);
752 for(int k=0; k<(1 << depth)+1; k++) {
753 diti_t tmp4 = inc_crd(tmp3, 2, k*del);
754 sample_point(tmp4, func);
755 }
756 }
757 }
758 // Centers
759 tmp = inc_all_crd(tmp, del/2);
760 for(int i=0; i<(1 << depth); i++) {
761 diti_t tmp2 = inc_crd(tmp, 0, i*del);
762 for(int j=0; j<(1 << depth); j++) {
763 diti_t tmp3 = inc_crd(tmp2, 1, j*del);
