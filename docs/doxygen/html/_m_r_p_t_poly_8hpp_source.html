249 bool addPointFromTree(typename treeT::iiCrdTupT iiCrdTup) {
250 /* If the point is already on the list, then return true */
251 if (polyDataGoodPoint2index.contains(iiCrdTup))
252 return true;
253 if ( !(tree.vertexExists(iiCrdTup))) {
254 std::cout << "ERROR(addPointFromTree): iiCrdTup is not a sampled point!" << std::endl;
255 return false;
256 }
257 /* Get the domain data associated with the point */
258 typename treeT::rangeT rPts = tree.getSample(iiCrdTup);
259 /* Make sure the point has no NaN's in it */
260 for(const auto index: polyDataPointDescription)
261 if ( (std::get<1>(index) == source::RANGE) && (std::isnan(tree.rngAt(rPts, std::get<2>(index)))) )
262 return false;
263 /* Make sure the point has no NaN's related to it in a scalar data set */
264 for(const auto index: polyDataScalarDescription)
265 if ( (std::get<1>(index) == source::RANGE) && (std::isnan(tree.rngAt(rPts, std::get<2>(index)))) )
266 return false;
267 /* Make sure the point has no NaN's related to it in a vector data set */
268 for(const auto index: polyDataVectorDescription)
269 if ( ((std::get<1>(index) == source::RANGE) && (std::isnan(tree.rngAt(rPts, std::get<2>(index))))) ||
270 ((std::get<3>(index) == source::RANGE) && (std::isnan(tree.rngAt(rPts, std::get<4>(index))))) ||
271 ((std::get<5>(index) == source::RANGE) && (std::isnan(tree.rngAt(rPts, std::get<6>(index))))) )
272 return false;
273 /* Make sure the point has no NaN's related to it in a color data set */
274 for(const auto index: polyDataColorDescription)
275 if ( ((std::get<1>(index) == source::RANGE) && (std::isnan(tree.rngAt(rPts, std::get<2>(index))))) ||
276 ((std::get<3>(index) == source::RANGE) && (std::isnan(tree.rngAt(rPts, std::get<4>(index))))) ||
277 ((std::get<5>(index) == source::RANGE) && (std::isnan(tree.rngAt(rPts, std::get<6>(index))))) )
278 return false;
279 /* Get the range data associated with the point */
280 typename treeT::domainT dPts = tree.iiCrdTup2domain(iiCrdTup);
281 /* Construct the new point array */
282 polyPointData newPoint;
283 for (int i=0; i<3; i++)
284 newPoint[i] = evaluateScalarDescriptor(polyDataPointDescription[i], dPts, rPts);
285 /* Compute the new points index -- i.e. an existing index if the point's physical coordinates are in an existing point, or a new number otherwise */
286 polyPointIndex newPointIndex = polyDataIndex2goodPoint.size();
287 if (polyPointUnique.contains(newPoint))
288 newPointIndex = polyPointUnique[newPoint];
289 /* Add everything to the data structures & return true*/
290 polyDataIndex2goodPoint.push_back(iiCrdTup);
291 polyDataGoodPoint2index[iiCrdTup] = newPointIndex;
292 return true;
293 }

454 int constructGeometryNonNaN(polyTypes polyType) {
455 if (polyDataPointDescription.size() == 0) {
456 std::cout << "ERROR: polyDataPointDescription must not be empty" << std::endl;
457 return -3;
458 }
459 /* Check polyType support */
460 switch(polyType) {
461 POINTS : ; break;
462 TRIANGLES :
463 if (tree.domainDimension != 2) {
464 std::cout << "ERROR: polyTypes::TRIANGLES is only supported when domain dimension is 2" << std::endl;
465 return -1;
466 }
467 break;
468 RECTANGLES :
469 if (tree.domainDimension != 2) {
470 std::cout << "ERROR: polyTypes::RECTANGLES is only supported when domain dimension is 2" << std::endl;
471 return -2;
472 }
473 break;
474 FANS :
475 if (tree.domainDimension != 2) {
476 std::cout << "ERROR: polyTypes::FANS is only supported when domain dimension is 2" << std::endl;
477 return -1;
478 }
479 break;
480 }
481 /* Set descriptive members */
482 polydataType = polyType;
483 /* Now we get a list of domain points for which we have good data */
484 polyDataIndex2goodPoint.clear();
485 polyDataGoodPoint2index.clear();
486 /* Construct good points (without NaNs in unpleasent places */
487 for (auto itr = tree.cbeginSamples(); itr != tree.cendSamples(); ++itr) {
488 addPointFromTree(itr->first);
489 }
490 /* Now construct poly cells */
491 polyDataGoodCells.clear();
492 if (polydataType == polyTypes::POINTS) {
493 for (auto vert: polyDataIndex2goodPoint)
494 addPolyFromTree(typename treeT::listCrdTup({vert}));
495 } else if (polydataType == polyTypes::FANS) {
496 typename treeT::listCrdTup cells = tree.getLeafCells(tree.getTopCell());
497 for(auto& cell: cells) {
498 for(int i=0; i<2; i++) {
499 for(int j=-1; j<2; j+=2) {
500 typename treeT::listCrdTup nbrs = tree.getExistingNeighbor(cell, i, j);
501 if (nbrs.size() > 1) {
502 for(auto n: nbrs) {
503 typename treeT::listCrdTup verts = tree.getCorners(n, i, -j);
504 addPolyFromTree(typename treeT::listCrdTup({verts[0], verts[1], cell}));
505 }
506 } else {
507 typename treeT::listCrdTup verts = tree.getCorners(cell, i, j);
508 addPolyFromTree(typename treeT::listCrdTup({verts[0], verts[1], cell}));
509 }
510 }
511 }
512 }
513 } else if (polydataType == polyTypes::TRIANGLES) {
514 typename treeT::listCrdTup cells = tree.getLeafCells(tree.getTopCell());
515 std::vector<int> omp({0, 1, 3, 2});
516 for(auto& cell: cells) {
517 typename treeT::listCrdTup verts = tree.getVertexes(cell);
518 for(int i=0; i<4; i++)
519 addPolyFromTree(typename treeT::listCrdTup({verts[omp[i]], verts[omp[(i+1)%4]], verts[4]}));
520 }
521 } else if (polydataType == polyTypes::RECTANGLES) {
522 typename treeT::listCrdTup cells = tree.getLeafCells(tree.getTopCell());
523 for(auto& cell: cells) {
524 typename treeT::listCrdTup verts = tree.getCorners(cell);
525 addPolyFromTree(typename treeT::listCrdTup({verts[0], verts[1], verts[3], verts[2]}));
526 }
527 }
528 /* All done. Return the number of good points. */
529 return polyDataIndex2goodPoint.size();
530 }

540 int writeVTK(std::string fileName) {
541 if (polyDataPointDescription.size() != 3) {
542 std::cout << "ERROR(writeVTK): polyDataPointDescription must contain precisely three scalarSourceDescriptores!" << std::endl;
543 return -3;
544 }
545 /* Looks like we have data. Let's open our file */
546 std::ofstream outStream;
547 outStream.open(fileName, std::ios::out | std::ios::binary | std::ios::trunc);
548 if (outStream.is_open()) {
549 outStream.imbue(std::locale::classic());
550 } else {
551 return 3;
552 }
553 /* Dump the header */
554 outStream << "# vtk DataFile Version 3.0" << std::endl;
555 outStream << "datasetname" << std::endl;
556 outStream << "ASCII" << std::endl;
557 outStream << "DATASET POLYDATA" << std::endl;
558 /* Dump the points */
559 outStream << "POINTS " << polyDataIndex2goodPoint.size() << " double" << std::endl;
560 for (const auto& k : polyDataIndex2goodPoint) {
561 typename treeT::domainT dPts = tree.iiCrdTup2domain(k);
562 typename treeT::rangeT rPts = tree.getSample(k);
563 std::vector<typename treeT::domainRangeBaseT> outPnt;
564 for (const auto& index : polyDataPointDescription) {
565 outPnt.push_back(evaluateScalarDescriptor(index, dPts, rPts));
566 }
567 outStream << std::setprecision(10) << outPnt[0] << " " << outPnt[1] << " " << outPnt[2] << std::setprecision(10) << std::endl;
568 }
569 /* Dump the cell data */
570 if (polydataType == polyTypes::POINTS) {
571 outStream << "VERTICES" << " " << polyDataIndex2goodPoint.size() << " " << (2*polyDataIndex2goodPoint.size()) << std::endl;
572 } else if ((polydataType == polyTypes::TRIANGLES) || (polydataType == polyTypes::RECTANGLES) || (polydataType == polyTypes::FANS)) {
573 if (tree.domainDimension == 1)
574 outStream << "LINES" << " " << polyDataGoodCells.size() << " " << ((1+polyDataGoodCells[0].size()) * polyDataGoodCells.size()) << std::endl;
575 else
576 outStream << "POLYGONS" << " " << polyDataGoodCells.size() << " " << ((1+polyDataGoodCells[0].size()) * polyDataGoodCells.size()) << std::endl;
577 }
578 for(auto& poly: polyDataGoodCells) {
579 outStream << poly.size() << " ";
580 for(auto& vert: poly) {
581 outStream << vert << " ";
582 }
583 outStream << std::endl;
584 }
585 /* Dump point scalar data */
586 if (polyDataScalarDescription.size() > 0) {
587 outStream << "POINT_DATA" << " " << polyDataIndex2goodPoint.size() << std::endl;
588 for (const auto& index : polyDataScalarDescription) {
589 outStream << "SCALARS " << std::get<0>(index) << " double 1" << std::endl;
590 outStream << "LOOKUP_TABLE default" << std::endl;
591 for (const auto& k : polyDataIndex2goodPoint) {
592 typename treeT::domainT dPts = tree.iiCrdTup2domain(k);
593 typename treeT::rangeT rPts = tree.getSample(k);
594 outStream << evaluateScalarDescriptor(index, dPts, rPts) << std::endl;
595 }
596 }
597 }
598
599
600
601 /* Dump point vector data */
602 if (polyDataVectorDescription.size() > 0) {
603 for (const auto& index : polyDataVectorDescription) {
604 outStream << "VECTORS " << std::get<0>(index) << " double" << std::endl;
605 for (const auto& k : polyDataIndex2goodPoint) {
606 typename treeT::domainT dPts = tree.iiCrdTup2domain(k);
607 typename treeT::rangeT rPts = tree.getSample(k);
608 polyPointData tmp = evaluateVectorDescriptor(index, dPts, rPts);
609 outStream << tmp[0] << " " << tmp[1] << " " << tmp[2] << std::endl;
610 }
611 }
612 }

618 outStream << std::endl;
619 return 0;
